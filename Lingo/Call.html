def talk(str)
  config.stdin.reopen(str)

  start

  %w[stdout stderr].flat_map { |key|
    io = config.send(key).tap(&:rewind)
    io.readlines.each(&:chomp!).tap {
      io.truncate(0)
      io.rewind
    }
  }.tap { |res|
    if block_given?
      res.map!(&Proc.new)
    else
      res.sort!
      res.uniq!
    end
  }
end
