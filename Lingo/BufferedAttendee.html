def forward_buffer
  # Aufgeschobene Einfügungen in Buffer kopieren

  @deferred_inserts.sort_by { |ins| ins.position }.each { |ins|
    @buffer.insert(ins.position, ins.object)
  }
  @deferred_inserts.clear

  # Buffer weiterleiten

  @buffer.each { |obj| forward(obj) }
  @buffer.clear
end
